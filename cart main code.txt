<script>
  const products = [
    {
      id: 1,
      name: "E Commerce Tech",
      description: "A complete solution for online store integration.",
      price: 50,
      originalPrice: 55,
      discount: 9,
      image: "https://img.freepik.com/premium-photo/ecommerce-concept-flat-isometric-vector-illustration_734610-17547.jpg"
    },
    {
      id: 2,
      name: "School Tech",
      description: "Empowering digital education platforms and tools.",
      price: 60,
      originalPrice: 65,
      discount: 8,
      image: "https://img.freepik.com/premium-photo/3d-flat-icon-environmental-education-concept-books-apps-focused-teaching-sustainability-with_980716-69679.jpg"
    }
  ];

  const cart = {};

  // Restore cart from sessionStorage if exists
  const storedCart = JSON.parse(sessionStorage.getItem("cart"));
  if (storedCart) {
    Object.assign(cart, storedCart);
  }

  function updateCartBar() {
    const cartBar = document.getElementById('cartBar');
    const cartCount = document.getElementById('cartCount');
    const cartSavings = document.getElementById('cartSavings');
    const cartIcon = document.getElementById('cartIcon');

    const totalItems = Object.values(cart).reduce((sum, item) => sum + item.quantity, 0);
    const totalSavings = Object.values(cart).reduce((sum, item) => sum + ((item.originalPrice - item.price) * item.quantity), 0);

    if (totalItems > 0) {
      cartBar.classList.remove('d-none');
      cartCount.textContent = `${totalItems} Item(s)`;
      cartSavings.textContent = `You save ₹${totalSavings}`;

      cartIcon.classList.add('cart-bounce');
      setTimeout(() => cartIcon.classList.remove('cart-bounce'), 500);
    } else {
      cartBar.classList.add('d-none');
    }
  }

  function renderProducts() {
    const container = document.getElementById('productContainer');
    container.innerHTML = '';

    products.forEach((p) => {
      const col = document.createElement('div');
      col.className = 'col-md-6 mb-4 d-flex';

      const card = document.createElement('div');
      card.className = 'card product-card flex-fill';

      card.innerHTML = `
        <img src="${p.image}" class="card-img-top" alt="${p.name}">
        <div class="card-body d-flex flex-column justify-content-between">
          <div>
            <h5 class="card-title font-weight-bold">${p.name}</h5>
            <p class="text-muted small">${p.description}</p>
            <h6>₹${p.price} <s class="text-muted">₹${p.originalPrice}</s>
            <span class="badge badge-warning ml-2">${p.discount}% OFF</span></h6>
          </div>
          <div id="product-controls-${p.id}" class="action-container mt-3">
            <button class="btn btn-add" onclick="addToCart(${p.id})">Add</button>
          </div>
        </div>
      `;

      col.appendChild(card);
      container.appendChild(col);
    });
  }

  function addToCart(id) {
    const product = products.find(p => p.id === id);
    if (!cart[id]) {
      cart[id] = { ...product, quantity: 1 };
    }
    sessionStorage.setItem("cart", JSON.stringify(cart));
    updateProductControls(id);
    updateCartBar();

    const card = document.querySelector(`#product-controls-${id}`).closest('.product-card');
    card.classList.add('glow-effect');
    setTimeout(() => card.classList.remove('glow-effect'), 800);
  }

  function increaseQuantity(id) {
    cart[id].quantity++;
    sessionStorage.setItem("cart", JSON.stringify(cart));
    updateProductControls(id);
    updateCartBar();
  }

  function decreaseQuantity(id) {
    if (cart[id].quantity > 1) {
      cart[id].quantity--;
    } else {
      delete cart[id];
    }
    sessionStorage.setItem("cart", JSON.stringify(cart));
    updateProductControls(id);
    updateCartBar();
  }

  function updateProductControls(id) {
    const container = document.getElementById(`product-controls-${id}`);
    const item = cart[id];

    if (!item) {
      container.innerHTML = `<button class="btn btn-add w-100" onclick="addToCart(${id})">Add</button>`;
    } else {
      container.innerHTML = `
        <div class="quantity-controls w-100">
          <button onclick="decreaseQuantity(${id})">−</button>
          <span>${item.quantity}</span>
          <button onclick="increaseQuantity(${id})">+</button>
        </div>
      `;
    }
  }

  document.addEventListener("DOMContentLoaded", () => {
    renderProducts();

    // After rendering, restore control buttons from cart
    if (storedCart) {
      Object.keys(cart).forEach(id => {
        updateProductControls(id);
      });
      updateCartBar();
    }
  });
</script>

</body>

<body>
  <div class="container mt-5">
    <h2 class="text-center mb-4">Checkout</h2>
    <div id="cartContainer"></div>

    <div id="emptyCartMessage" style="display: none;">
      <h4>Your cart is empty</h4>
      <a href="index.html" class="btn btn-primary back-link">Back to Products</a>
    </div>
  </div>

  <script>
    
    const cart = JSON.parse(sessionStorage.getItem("cart")) || {};

    function renderCart() {
      const container = document.getElementById("cartContainer");
      container.innerHTML = "";

      const items = Object.values(cart);
      if (items.length === 0) {
        container.style.display = "none";
        document.getElementById("emptyCartMessage").style.display = "block";
        return;
      }

      let totalItems = 0, totalPrice = 0, totalSaved = 0;

      items.forEach((item, index) => {
        totalItems += item.quantity;
        totalPrice += item.price * item.quantity;
        totalSaved += (item.originalPrice - item.price) * item.quantity;

        const row = document.createElement("div");
        row.className = "cart-item d-flex align-items-center justify-content-between";

        row.innerHTML = `
          <div class="d-flex align-items-center">
            <img src="${item.image}" alt="${item.name}">
            <div>
              <h5>${item.name}</h5>
              <p class="mb-1">Qty: ${item.quantity} × ₹${item.price}</p>
              <p class="mb-0 text-success">You save ₹${(item.originalPrice - item.price) * item.quantity}</p>
            </div>
          </div>
          <span class="remove-btn" onclick="removeItem(${index})">&times;</span>
        `;

        container.appendChild(row);
      });

      const summary = document.createElement("div");
      summary.className = "summary mt-4";

      summary.innerHTML = `
        <p><strong>Total Items:</strong> ${totalItems}</p>
        <p><strong>Total Price:</strong> ₹${totalPrice}</p>
        <p class="text-success"><strong>You Saved:</strong> ₹${totalSaved}</p>
        <button class="btn btn-primary mb-2" onclick="placeOrder()">Place Order</button>
        <a href="index.html" class="btn btn-outline-secondary">Back to Products</a>
      `;

      container.appendChild(summary);
    }

    function removeItem(index) {
      const keys = Object.keys(cart);
      delete cart[keys[index]];
      sessionStorage.setItem("cart", JSON.stringify(cart));
      renderCart();
    }

    function placeOrder() {
      if (Object.keys(cart).length === 0) return;
      window.location.href = "payment.html";
    }

    renderCart();
  </script>
</body>
</html>
